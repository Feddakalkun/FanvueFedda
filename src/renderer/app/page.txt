'use client';

import { useState, useEffect } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import {
  Plus,
  Settings,
  LayoutGrid,
  Zap,
  Calendar,
  Cloud,
  Search,
  CheckCircle2,
  X,
  RefreshCw,
  AlertCircle,
  Sparkles,
  User,
  Image as ImageIcon,
  Video,
  Mic,
  Home as HomeIcon,
  Cpu,
  Smartphone,
  MessageSquare,
  Lock,
  Globe,
  Rocket,
  ChevronRight,
  ShieldCheck,
  Instagram,
  Twitter,
  Music2,
  Shuffle,
  Trash2
} from 'lucide-react';
import { cn } from '@/lib/utils';
import { useRouter } from 'next/navigation';

export default function Home() {
  const [view, setView] = useState<'landing' | 'studio'>('landing');
  const [activeTab, setActiveTab] = useState('Workspace');
  const [isCreating, setIsCreating] = useState(false);
  const [isLoading, setIsLoading] = useState(true);

  interface NavItem {
    name: string;
    icon: any;
    onClick?: () => void;
  }

  // App Config & User State
  const [characters, setCharacters] = useState<any[]>([]);
  const [activeCharacter, setActiveCharacter] = useState<any>(null);
  const [isEngineOnline, setIsEngineOnline] = useState(false);
  const [isSettingsOpen, setIsSettingsOpen] = useState(false);
  const [globalConfig, setGlobalConfig] = useState({
    fanvueClientId: '',
    fanvueClientSecret: ''
  });
  const [previewImage, setPreviewImage] = useState<string | null>(null);
  const [isGenerating, setIsGenerating] = useState(false);
  const [isCaptioning, setIsCaptioning] = useState(false);
  const [modelPackages, setModelPackages] = useState<any[]>([]);
  const [downloadLog, setDownloadLog] = useState('');
  const [isDownloading, setIsDownloading] = useState(false);
  const [activeSettingsTab, setActiveSettingsTab] = useState<'General' | 'Models'>('General');
  const [systemStats, setSystemStats] = useState<any>(null);
  const [availableLoras, setAvailableLoras] = useState<string[]>([]);
  const [selectedLoras, setSelectedLoras] = useState<{ name: string, strength: number }[]>([]);
  const [targetStrength, setTargetStrength] = useState(1.2);
  const [studioMode, setStudioMode] = useState<'image' | 'video' | 'swap'>('image');
  const [sessionAssets, setSessionAssets] = useState<{ type: 'image' | 'video', url: string, timestamp: number }[]>([]);
  const [isSwapping, setIsSwapping] = useState(false);
  const [swapPrompt, setSwapPrompt] = useState('');

  // New Character Form State
  const [newPersona, setNewPersona] = useState({
    name: '',
    handle: '',
    bio: '',
    loraMix: '',
    looksDescription: '',
    attributes: {
      vibe: '',
      ethnicity: '',
      skinTone: '',
      hairStyle: '',
      bodyType: '',
      breastSize: '',
    },
    platformSettings: {
      tiktok: { username: '', password: '', googleAuth: false, enabled: false },
      insta: { username: '', password: '', googleAuth: false, enabled: false },
      x: { username: '', password: '', googleAuth: false, enabled: false }
    }
  });

  const handleCreatePersona = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    try {
      const res = await fetch('/api/characters', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(newPersona)
      });
      if (res.ok) {
        const data = await res.json();
        setCharacters(prev => [data, ...prev]);
        setActiveCharacter(data);
        setIsCreating(false);
        setActiveTab('Workspace');
      } else {
        const errorData = await res.json().catch(() => ({}));
        alert(`Failed to save character: ${errorData.error || 'Check server logs'}`);
      }
    } catch (e) {
      alert('Communication error');
    } finally {
      setIsLoading(false);
      window.location.reload();
    }
  };

  useEffect(() => {
    async function init() {
      setIsLoading(true);
      try {
        // Fetch Characters
        const charRes = await fetch('/api/characters');
        const charData = await charRes.json();
        setCharacters(charData);
        if (charData.length > 0) setActiveCharacter(charData[0]);

        // Fetch Global Config
        const configRes = await fetch('/api/config');
        const configData = await configRes.json();
        if (configData) {
          setGlobalConfig({
            fanvueClientId: configData.fanvueClientId || '',
            fanvueClientSecret: configData.fanvueClientSecret || ''
          });
        }
      } catch (e) {
        console.error('Core failed to initialize');
      } finally {
        setIsLoading(false);
      }
    }
    init();

    // Check ComfyUI Status & Stats
    const checkEngine = async () => {
      try {
        const res = await fetch('http://localhost:8188/system_stats');
        if (res.ok) {
          const stats = await res.json();
          setSystemStats(stats);
          setIsEngineOnline(true);
        } else {
          setIsEngineOnline(false);
        }
      } catch {
        setIsEngineOnline(false);
      }
    };
    checkEngine();
    const interval = setInterval(checkEngine, 5000);

    fetchModelStatus();
    fetchLoras();

    return () => clearInterval(interval);
  }, []);

  const fetchLoras = async () => {
    try {
      const res = await fetch('/api/models/loras');
      const data = await res.json();
      setAvailableLoras(data);
    } catch (e) {
      console.error('Failed to fetch loras');
    }
  };

  const fetchModelStatus = async () => {
    try {
      const res = await fetch('/api/models');
      const data = await res.json();
      setModelPackages(data);
    } catch (e) {
      console.error('Failed to fetch models');
    }
  };

  useEffect(() => {
    let pollInterval: any;
    if (isSettingsOpen || isDownloading) {
      pollInterval = setInterval(async () => {
        const res = await fetch('/api/models/download');
        const data = await res.json();
        setDownloadLog(data.log);
        setIsDownloading(data.isDownloading);
        if (!data.isDownloading && isDownloading) {
          fetchModelStatus();
        }
      }, 2000);
    }
    return () => clearInterval(pollInterval);
  }, [isSettingsOpen, isDownloading]);

  const handleStartDownload = async (packageId: string) => {
    try {
      const res = await fetch('/api/models/download', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ packageId })
      });
      if (res.ok) setIsDownloading(true);
    } catch (e) {
    }
  };

  const handlePurgeVRAM = async () => {
    try {
      await fetch('http://localhost:8188/free', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ unload_models: true, free_memory: true })
      });
      alert('VRAM Purged');
    } catch (e) {
      alert('Failed to purge VRAM');
    }
  };

  const handleGeneratePreview = async () => {
    const loraMixString = selectedLoras
      .map(lora => `${lora.name}:${lora.strength}`)
      .join(', ');

    if (!loraMixString && !newPersona.loraMix) return alert('Please select at least one LoRA');

    setIsGenerating(true);
    try {
      const res = await fetch('/api/generate/portrait', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...newPersona,
          loraMix: loraMixString || newPersona.loraMix
        })
      });
      const data = await res.json();
      if (data.image) setPreviewImage(data.image);
      else alert('Generation failed: ' + (data.error || 'Unknown error'));
    } catch (e) {
      alert('Failed to connect to AI engine');
    } finally {
      setIsGenerating(false);
    }
  };

  const handleGenerateVideo = async () => {
    if (!previewImage || !activeCharacter) return alert('Select a portrait first');
    setIsGenerating(true);
    try {
      const res = await fetch('/api/generate/video', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          imageBase64: previewImage,
          persona: activeCharacter,
          workflow: studioMode === 'video' ? 'wan_i2v' : 'ltx_lipsync'
        })
      });
      const data = await res.json();
      if (data.video) {
        setSessionAssets(prev => [{ type: 'video', url: data.video, timestamp: Date.now() }, ...prev]);
      } else alert('Video generation failed');
    } catch (e) {
      alert('Failed to connect to video cluster');
    } finally {
      setIsGenerating(false);
    }
  };

  const handleClothesSwap = async () => {
    if (!previewImage || !swapPrompt) return alert('Describe the new outfit');
    setIsSwapping(true);
    try {
      const res = await fetch('/api/generate/swap', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          imageBase64: previewImage,
          prompt: swapPrompt,
          persona: activeCharacter
        })
      });
      const data = await res.json();
      if (data.image) {
        setPreviewImage(data.image);
        setSessionAssets(prev => [{ type: 'image', url: data.image, timestamp: Date.now() }, ...prev]);
      } else alert('Swap failed');
    } catch (e) {
      alert('Failed to connect to swap engine');
    } finally {
      setIsSwapping(false);
    }
  };

  const handleCaptionImage = async () => {
    if (!previewImage) return;
    setIsCaptioning(true);
    try {
      const res = await fetch('/api/generate/caption', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ imageBase64: previewImage })
      });
      const data = await res.json();
      if (data.caption) {
        setNewPersona(prev => ({ ...prev, looksDescription: data.caption }));
      } else {
        alert('Captioning failed: ' + (data.error || 'Unknown error'));
      }
    } catch (e) {
      alert('Failed to connect to captioning service');
    } finally {
      setIsCaptioning(false);
    }
  };

  const handleRandomizeMix = () => {
    if (selectedLoras.length === 0) return;

    // Generate random raw weights
    const rawWeights = selectedLoras.map(() => Math.random());
    const sum = rawWeights.reduce((a, b) => a + b, 0);

    // Scale to targetStrength
    const newMix = selectedLoras.map((lora, i) => ({
      ...lora,
      strength: parseFloat(((rawWeights[i] / sum) * targetStrength).toFixed(2))
    }));

    setSelectedLoras(newMix);
  };

  const handleDeleteCharacter = async (id: string) => {
    if (!confirm('Are you sure you want to delete this character?')) return;
    try {
      const res = await fetch(`/api/characters?id=${id}`, { method: 'DELETE' });
      if (res.ok) {
        setCharacters(prev => prev.filter(c => c.id !== id));
        if (activeCharacter?.id === id) setActiveCharacter(null);
      }
    } catch (e) {
      alert('Delete failed');
    }
  };

  const navItems: NavItem[] = [
    { name: 'Workspace', icon: LayoutGrid },
    { name: 'AI Studio', icon: Zap },
    { name: 'Socials', icon: Globe },
    { name: 'Settings', icon: Settings, onClick: () => setIsSettingsOpen(true) },
  ];

  const handleSaveSettings = async () => {
    try {
      await fetch('/api/config', {
        method: 'POST',
        body: JSON.stringify(globalConfig)
      });
      setIsSettingsOpen(false);
    } catch (e) {
      alert('Failed to save configuration');
    }
  };

  // --- RENDERING ---

  if (isLoading) {
    return (
      <div className="min-h-screen bg-black flex items-center justify-center">
        <RefreshCw className="w-8 h-8 text-white/20 animate-spin" />
      </div>
    );
  }

  if (view === 'landing') {
    return (
      <div className="relative min-h-screen bg-[#050505] flex flex-col items-center justify-center overflow-hidden font-sans">
        <div className="absolute inset-0 z-0 opacity-20">
          <div className="absolute inset-0 bg-gradient-to-b from-transparent via-[#050505]/80 to-[#050505] z-10"></div>
          <div className="w-full h-full bg-[url('https://images.unsplash.com/photo-1618005182384-a83a8bd57fbe?q=80&w=1964&auto=format&fit=crop')] bg-cover bg-center grayscale brightness-50"></div>
        </div>

        <motion.div
          initial={{ opacity: 0, y: 30 }}
          animate={{ opacity: 1, y: 0 }}
          className="relative z-10 flex flex-col items-center text-center gap-16"
        >
          <div className="space-y-4">
            <h1 className="text-7xl md:text-9xl font-black tracking-[0.2em] text-white italic underline decoration-white/5 underline-offset-8">FEDDAKALKUN</h1>
          </div>

          <motion.button
            whileHover={{ scale: 1.05, backgroundColor: '#fff', color: '#000' }}
            whileTap={{ scale: 0.95 }}
            onClick={() => setView('studio')}
            className="group relative flex items-center gap-6 px-20 py-8 border-2 border-white/10 rounded-full overflow-hidden transition-all duration-500"
          >
            <div className="absolute inset-x-0 bottom-0 h-0 bg-white group-hover:h-full transition-all duration-500 -z-10"></div>
            <span className="text-white group-hover:text-black font-black uppercase tracking-[0.5em] text-xs">ENTER</span>
            <ChevronRight className="w-5 h-5 text-white group-hover:text-black group-hover:translate-x-2 transition-all" />
          </motion.button>
        </motion.div>
      </div>
    );
  }

  return (
    <>
      <div className="min-h-screen bg-black text-white font-sans selection:bg-white/20">
        {/* Universal Desktop Top Bar */}
        <header className="border-b border-white/5 bg-black/60 backdrop-blur-2xl sticky top-0 z-50">
          <div className="max-w-[1600px] mx-auto px-10 h-20 flex items-center justify-between">
            <div className="flex items-center gap-12 h-full">
              <div className="font-black italic text-xl tracking-tighter">FEDDAKALKUN</div>
              <nav className="flex items-center gap-8 h-full">
                {navItems.map((item) => (
                  <button
                    key={item.name}
                    onClick={() => {
                      if (item.onClick) item.onClick();
                      else setActiveTab(item.name);
                    }}
                    className={cn(
                      "relative h-full px-2 text-[11px] font-black uppercase tracking-widest transition-all transition-colors flex items-center gap-2",
                      activeTab === item.name ? "text-white" : "text-white/30 hover:text-white/60"
                    )}
                  >
                    <item.icon className="w-3.5 h-3.5" />
                    {item.name}
                    {activeTab === item.name && (
                      <motion.div layoutId="navGlow" className="absolute bottom-0 left-0 right-0 h-0.5 bg-white rounded-full shadow-[0_-5px_15px_rgba(255,255,255,0.5)]" />
                    )}
                  </button>
                ))}
              </nav>
            </div>

            <div className="flex items-center gap-6">
              <div className="flex flex-col items-end">
                <div className="flex items-center gap-3 bg-white/5 px-4 py-2 rounded-full border border-white/10">
                  <div className={cn("w-2 h-2 rounded-full animate-pulse", isEngineOnline ? "bg-green-500 shadow-[0_0_10px_#22c55e]" : "bg-red-500 shadow-[0_0_10px_#ef4444]")}></div>
                  <span className="text-[10px] font-black uppercase tracking-widest text-white/50 italic">
                    ComfyUI: {isEngineOnline ? 'Online' : 'Offline'}
                  </span>
                </div>
                {isEngineOnline && systemStats && (
                  <div className="flex gap-4 mt-2 px-2">
                    <span className="text-[8px] font-black uppercase tracking-widest text-white/20 italic">
                      VRAM: {Math.round((systemStats.devices[0]?.vram_total - systemStats.devices[0]?.vram_free) / 1024 / 1024 / 1024)}GB / {Math.round(systemStats.devices[0]?.vram_total / 1024 / 1024 / 1024)}GB
                    </span>
                  </div>
                )}
              </div>
            </div>
          </div>
        </header>

        <main className="max-w-[1900px] mx-auto p-8 lg:p-12">
          <AnimatePresence mode="wait">
            {activeTab === 'AI Studio' ? (
              <>
                <motion.div key="studio" initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: -20 }} className="h-[80vh] grid grid-cols-1 lg:grid-cols-12 gap-10">
                  {/* Studio Sidebar: Character Quick Select */}
                  <div className="lg:col-span-3 space-y-8 h-full overflow-y-auto pr-4 custom-scrollbar">
                    <div className="bg-white/5 p-8 rounded-[40px] border border-white/10 space-y-6">
                      <h3 className="text-[10px] font-black uppercase tracking-[0.3em] text-white/30 italic">Select Persona</h3>
                      <div className="space-y-4">
                        {characters.map(c => (
                          <button
                            key={c.id}
                            onClick={() => setActiveCharacter(c)}
                            className={cn(
                              "w-full flex items-center gap-6 p-4 rounded-2xl border transition-all group",
                              activeCharacter?.id === c.id ? "bg-white border-white text-black shadow-[0_10px_30px_rgba(255,255,255,0.1)]" : "bg-white/5 border-white/5 text-white/40 hover:bg-white/10"
                            )}
                          >
                            <div className={cn("w-10 h-10 rounded-xl flex items-center justify-center font-black italic", activeCharacter?.id === c.id ? "bg-black text-white" : "bg-white/10")}>
                              {c.name.charAt(0)}
                            </div>
                            <div className="text-left">
                              <p className="text-[11px] font-black uppercase">{c.name}</p>
                              <p className="text-[8px] font-bold opacity-40 italic">{c.handle}</p>
                            </div>
                          </button>
                        ))}
                      </div>
                    </div>

                    {/* Mode Selector */}
                    <div className="grid grid-cols-3 gap-2 p-2 bg-white/5 rounded-3xl border border-white/10">
                      {(['image', 'video', 'swap'] as const).map(mode => (
                        <button
                          key={mode}
                          onClick={() => setStudioMode(mode)}
                          className={cn("py-3 rounded-2xl text-[8px] font-black uppercase tracking-widest transition-all", studioMode === mode ? "bg-white text-black" : "text-white/30 hover:text-white")}
                        >
                          {mode}
                        </button>
                      ))}
                    </div>
                  </div>

                  {/* Studio Canvas: The Creator */}
                  <div className="lg:col-span-6 flex flex-col gap-10">
                    <div className="relative aspect-[3/4] bg-white/[0.02] border border-white/5 rounded-[64px] overflow-hidden group shadow-2xl">
                      {previewImage ? (
                        <img src={previewImage} className="w-full h-full object-cover" />
                      ) : (
                        <div className="absolute inset-0 flex flex-col items-center justify-center gap-4 text-white/10">
                          <ImageIcon className="w-12 h-12" />
                          <span className="text-[10px] font-black uppercase tracking-[0.4em]">Awaiting Generation</span>
                        </div>
                      )}
                      {isGenerating && (
                        <div className="absolute inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center z-20">
                          <div className="flex flex-col items-center gap-6">
                            <RefreshCw className="w-10 h-10 text-white animate-spin" />
                            <span className="text-[10px] font-black uppercase tracking-[0.4em] italic animate-pulse">Processing Masterframe...</span>
                          </div>
                        </div>
                      )}
                    </div>

                    {/* Film Strip Gallery */}
                    <div className="bg-white/5 h-32 rounded-[32px] border border-white/10 p-4 flex gap-4 overflow-x-auto custom-scrollbar">
                      {sessionAssets.length === 0 && <div className="flex-1 flex items-center justify-center text-[8px] font-black uppercase tracking-widest text-white/10">Empty Session Strip</div>}
                      {sessionAssets.map((asset, i) => (
                        <button key={i} onClick={() => setPreviewImage(asset.url)} className="flex-shrink-0 h-full aspect-[3/4] rounded-lg overflow-hidden border border-white/10 hover:border-white transition-all">
                          {asset.type === 'video' ? <div className="w-full h-full bg-blue-500/20 flex items-center justify-center"><Video className="w-4 h-4" /></div> : <img src={asset.url} className="w-full h-full object-cover" />}
                        </button>
                      ))}
                    </div>
                  </div>

                  {/* Studio Controls: Demographics & Styling */}
                  <div className="lg:col-span-3 space-y-8 h-full overflow-y-auto pr-4 custom-scrollbar">
                    {studioMode === 'swap' ? (
                      <div className="bg-white/5 p-8 rounded-[40px] border border-white/10 space-y-8">
                        <h3 className="text-[10px] font-black uppercase tracking-[0.3em] text-white/30 italic">Virtual Try-On</h3>
                        <textarea
                          value={swapPrompt}
                          onChange={e => setSwapPrompt(e.target.value)}
                          placeholder="Describe the new outfit... (e.g. skin-tight black latex bodysuit, micro-mesh bikini, oversized white hoodie)"
                          className="w-full bg-black border border-white/5 rounded-2xl p-6 text-[10px] font-medium italic min-h-[150px] outline-none focus:border-white/20 transition-all text-white/80"
                        />
                        <button
                          onClick={handleClothesSwap}
                          disabled={isSwapping || !previewImage}
                          className="w-full bg-white text-black py-5 rounded-2xl font-black uppercase tracking-[0.3em] text-[10px] hover:invert transition-all disabled:opacity-50"
                        >
                          {isSwapping ? 'Swapping Outfit...' : 'Perform Swap'}
                        </button>
                      </div>
                    ) : studioMode === 'video' ? (
                      <div className="bg-white/5 p-8 rounded-[40px] border border-white/10 space-y-8">
                        <h3 className="text-[10px] font-black uppercase tracking-[0.3em] text-white/30 italic">Video Cluster Config</h3>
                        <div className="space-y-4">
                          <div className="p-4 bg-black/40 rounded-2xl border border-white/5 space-y-3">
                            <span className="text-[8px] font-black text-white/20 uppercase">Model Choice</span>
                            <select className="w-full bg-transparent text-[10px] font-black uppercase outline-none text-white/60 appearance-none cursor-pointer">
                              <option>Wan 2.1 I2V (14B)</option>
                              <option>LTX-2 Distill (19B)</option>
                            </select>
                          </div>
                        </div>
                        <button
                          onClick={handleGenerateVideo}
                          disabled={isGenerating || !previewImage}
                          className="w-full bg-white text-black py-5 rounded-2xl font-black uppercase tracking-[0.3em] text-[10px] hover:invert transition-all disabled:opacity-50 flex items-center justify-center gap-4"
                        >
                          Breathe Life <Zap className="w-3.5 h-3.5 fill-current" />
                        </button>
                      </div>
                    ) : (
                      /* Standard Aesthetic Controls for Studio */
                      <div className="bg-white/5 p-8 rounded-[40px] border border-white/10 space-y-8 text-left">
                        <h3 className="text-[10px] font-black uppercase tracking-[0.3em] text-white/30 italic">Aesthetic Refinement</h3>
                        <div className="space-y-6">
                          <div className="space-y-3">
                            <label className="text-[8px] font-black text-white/20 uppercase ml-2 italic">Active Vibe</label>
                            <select
                              className="w-full bg-black border border-white/5 rounded-2xl p-4 text-[10px] font-black uppercase outline-none text-white/60"
                              value={activeCharacter?.attributes?.vibe}
                              onChange={e => setActiveCharacter((prev: any) => ({ ...prev, attributes: { ...prev.attributes, vibe: e.target.value } }))}
                            >
                              {[
                                'Scandi TikTok Replica 1 (Plaid Pants)',
                                'Scandi TikTok Replica 2 (White Corset)',
                                'Scandi TikTok Replica 3 (Lace Top)',
                                'Scandi TikTok Y2K',
                                'TikTok / Mirror Selfie',
                                'Flash / Night Out',
                                'Casual Indoor',
                                'Late Night Explicit (Lingerie)',
                                'BDSM Amateur (Choker)',
                                '[Grok] Bedroom Mirror (Low-Rise Jeans)',
                                '[Grok] Bathroom (Wrap Top + Leggings)',
                                '[Grok] Bedroom (Long-Sleeve V-Neck)',
                                '[Grok] Poster Wall (Lace Top)',
                                '[Grok] Mirror Selfie (Low-Rise Jeans)',
                                '[Grok] Casual Bedroom Dance',
                                '[Grok] Bathroom Confident Pose',
                                '[Grok] Bedroom Poster Wall Pose',
                                '[Grok] Bedroom Mirror Hair Flip',
                                '[Grok] Light Bedroom Dance',
                                '[Grok] Bed Lip-Sync',
                                '[Grok] Door Frame Confident Lean',

                        { name: 'Post Content', icon: Globe, desc: 'Sync to social channels' },
                        { name: 'Edit Character', icon: MessageSquare, desc: 'Modify details and settings' },
                        { name: 'Engine Config', icon: Cpu, desc: 'Adjust LoRA and prompts' }
                      ].map(action => (
                        <button key={action.name} className="glass-card p-10 bg-white/[0.01] hover:bg-white/[0.03] border-white/5 hover:border-white/10 !rounded-[48px] text-left transition-all group">
                          <div className="w-12 h-12 bg-white/5 rounded-2xl flex items-center justify-center mb-6 border border-white/10 group-hover:scale-110 group-hover:bg-white group-hover:text-black transition-all">
                            <action.icon className="w-5 h-5" />
                          </div>
                          <h4 className="font-black italic text-lg uppercase mb-2 leading-none">{action.name}</h4>
                          <p className="text-[10px] font-black uppercase tracking-widest text-white/20 italic">{action.desc}</p>
                        </button>
                      ))}
                    </div>

                    <div className="glass-card p-10 bg-white/[0.01] border-white/5 !rounded-[48px]">
                      <div className="flex items-center justify-between mb-8">
                        <h4 className="text-xs font-black uppercase tracking-[0.4em] text-white/30 italic">Social Connections</h4>
                        <RefreshCw className="w-4 h-4 text-white/10" />
                      </div>
                      <div className="grid grid-cols-4 gap-8">
                        {[
                          { id: 'fanvue', name: 'Fanvue', icon: Cloud, enabled: !!activeCharacter?.fanvueConnected },
                          { id: 'tiktok', name: 'TikTok', icon: Music2, enabled: activeCharacter?.platformSettings?.tiktok?.username?.length > 0 },
                          { id: 'insta', name: 'Instagram', icon: Instagram, enabled: activeCharacter?.platformSettings?.insta?.username?.length > 0 },
                          { id: 'x', name: 'Twitter', icon: Twitter, enabled: activeCharacter?.platformSettings?.x?.username?.length > 0 },
                        ].map(plat => (
                          <div key={plat.id} className="flex flex-col items-center gap-4 group">
                            <div className={cn(
                              "w-16 h-16 rounded-2xl flex items-center justify-center border transition-all duration-500",
                              plat.enabled ? "bg-white text-black border-white shadow-[0_10px_30px_rgba(255,255,255,0.2)]" : "bg-white/5 text-white/20 border-white/5 grayscale"
                            )}>
                              <plat.icon className="w-6 h-6" />
                            </div>
                            <span className={cn("text-[9px] font-black uppercase tracking-widest italic", plat.enabled ? "text-white" : "text-white/20")}>{plat.name}</span>
                            <div className={cn("w-1.5 h-1.5 rounded-full", plat.enabled ? "bg-green-500 shadow-[0_0_10px_#22c55e]" : "bg-red-500/20")}></div>
                          </div>
                        ))}
                      </div>
                    </div>
                  </div>
                </div>
              </motion.div>
            )}
          </AnimatePresence>
        </main>
      </div >

      {/* Global Settings Modal */}
      <AnimatePresence>
        {
          isSettingsOpen && (
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
              className="fixed inset-0 z-[100] flex items-center justify-center p-10 pointer-events-none"
            >
              <div className="absolute inset-0 bg-black/60 backdrop-blur-3xl pointer-events-auto" onClick={() => setIsSettingsOpen(false)}></div>
              <motion.div
                initial={{ scale: 0.9, opacity: 0, y: 20 }}
                animate={{ scale: 1, opacity: 1, y: 0 }}
                exit={{ scale: 0.9, opacity: 0, y: 20 }}
                className="relative w-full max-w-4xl bg-white/[0.02] border border-white/10 rounded-[48px] p-12 shadow-[0_40px_100px_rgba(0,0,0,0.5)] pointer-events-auto overflow-hidden"
              >
                <header className="flex justify-between items-center mb-10">
                  <div className="flex items-center gap-10">
                    <h2 className="text-3xl font-black italic uppercase tracking-tighter">Settings</h2>
                    <div className="flex bg-white/5 p-1 rounded-2xl border border-white/10">
                      {(['General', 'Models'] as const).map((tab) => (
                        <button
                          key={tab}
                          onClick={() => setActiveSettingsTab(tab)}
                          className={cn(
                            "px-6 py-2 rounded-xl text-[10px] font-black uppercase tracking-widest transition-all",
                            activeSettingsTab === tab ? "bg-white text-black" : "text-white/40 hover:text-white"
                          )}
                        >
                          {tab}
                        </button>
                      ))}
                    </div>
                  </div>
                  <button onClick={() => setIsSettingsOpen(false)} className="p-3 hover:bg-white/5 rounded-full transition-colors">
                    <X className="w-6 h-6 text-white/20" />
                  </button>
                </header>

                <div className="max-h-[60vh] overflow-y-auto pr-4 custom-scrollbar">
                  {activeSettingsTab === 'General' ? (
                    <div className="space-y-12">
                      <section className="space-y-8">
                        <div className="flex items-center gap-4 text-white/20">
                          <div className="w-8 h-8 bg-white/5 rounded-lg flex items-center justify-center border border-white/10 text-[10px] font-black">01</div>
                          <h3 className="text-xs font-black uppercase tracking-[0.3em]">Fanvue Developer Access</h3>
                        </div>

                        <div className="grid grid-cols-2 gap-8">
                          <div className="space-y-3">
                            <label className="text-[10px] font-black text-white/20 uppercase tracking-widest block ml-2">Client ID</label>
                            <input
                              type="password"
                              value={globalConfig.fanvueClientId}
                              onChange={(e) => setGlobalConfig({ ...globalConfig, fanvueClientId: e.target.value })}
                              className="w-full bg-black border border-white/5 rounded-2xl py-5 px-8 text-xs outline-none focus:border-white/30 transition-all font-mono"
                              placeholder="fv_client_..."
                            />
                          </div>
                          <div className="space-y-3">
                            <label className="text-[10px] font-black text-white/20 uppercase tracking-widest block ml-2">Client Secret</label>
                            <input
                              type="password"
                              value={globalConfig.fanvueClientSecret}
                              onChange={(e) => setGlobalConfig({ ...globalConfig, fanvueClientSecret: e.target.value })}
                              className="w-full bg-black border border-white/5 rounded-2xl py-5 px-8 text-xs outline-none focus:border-white/30 transition-all font-mono"
                              placeholder="fv_secret_..."
                            />
                          </div>
                        </div>
                      </section>
                    </div>
                  ) : (
                    <div className="space-y-12">
                      <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
                        <div className="space-y-6">
                          {modelPackages.map(pkg => (
                            <div key={pkg.id} className="p-8 border border-white/5 rounded-[40px] bg-white/[0.01] space-y-4 hover:border-white/10 transition-all">
                              <div className="flex justify-between items-start">
                                <div>
                                  <h4 className="font-black italic text-lg uppercase leading-none mb-1">{pkg.name}</h4>
                                  <p className="text-[9px] font-black uppercase tracking-widest text-white/20 italic">{pkg.size} • {pkg.description}</p>
                                </div>
                                {pkg.installed ? (
                                  <div className="bg-green-500/10 text-green-500 px-3 py-1 rounded-full text-[8px] font-black uppercase tracking-widest border border-green-500/20">Ready</div>
                                ) : (
                                  <button
                                    disabled={isDownloading}
                                    onClick={() => handleStartDownload(pkg.id)}
                                    className="bg-white text-black px-4 py-2 rounded-full text-[8px] font-black uppercase tracking-widest hover:brightness-90 transition-all disabled:opacity-50"
                                  >
                                    Download
                                  </button>
                                )}
                              </div>
                            </div>
                          ))}
                        </div>

                        <div className="space-y-4">
                          <div className="flex items-center justify-between">
                            <h4 className="text-[10px] font-black uppercase tracking-[0.4em] text-white/20 italic">Terminal Output</h4>
                            <div className="flex items-center gap-4">
                              <button
                                onClick={handlePurgeVRAM}
                                className="px-3 py-1 bg-red-500/10 text-red-500 hover:bg-red-500 hover:text-white border border-red-500/20 rounded-lg text-[8px] font-black uppercase tracking-widest transition-all italic"
                              >
                                Purge VRAM
                              </button>
                              {isDownloading && (
                                <div className="flex items-center gap-2">
                                  <div className="w-1.5 h-1.5 bg-white animate-pulse rounded-full"></div>
                                  <span className="text-[8px] font-black text-white uppercase tracking-widest italic">Syncing...</span>
                                </div>
                              )}
                            </div>
                          </div>
                          <div className="bg-black border border-white/5 rounded-[32px] p-6 h-[400px] overflow-y-auto font-mono text-[9px] selection:bg-white/20 custom-scrollbar whitespace-pre-wrap text-white/60">
                            {downloadLog || "Ready to sync models..."}
                          </div>
                        </div>
                      </div>
                    </div>
                  )}
                </div>

                <footer className="mt-12 pt-8 border-t border-white/5 flex justify-end gap-6">
                  <button
                    onClick={() => setIsSettingsOpen(false)}
                    className="px-10 py-4 border border-white/5 hover:bg-white/5 rounded-full text-[10px] font-black uppercase tracking-widest transition-all italic text-white/40 hover:text-white"
                  >
                    Close
                  </button>
                  <button
                    onClick={() => {
                      handleSaveSettings();
                      setIsSettingsOpen(false);
                    }}
                    className="bg-white text-black px-12 py-4 rounded-full text-[10px] font-black uppercase tracking-widest hover:invert transition-all active:scale-95 shadow-xl shadow-white/5 italic flex items-center gap-3"
                  >
                    Apply Changes
                  </button>
                </footer>
              </motion.div>
            </motion.div>
          )}
      </AnimatePresence>
    </>
  );
}
